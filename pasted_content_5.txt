// ==UserScript==
// @name         SIAP Automator - V8.4.6 (Sem índices nos textos)
// @namespace    http://tampermonkey.net/
// @version      8.4.6
// @description  Automação total com botão Parar. IA não insere índices em metodologia/avaliação.
// @match        https://siap.educacao.go.gov.br/*
// @match        https://*.siap.com.br/*
// @match        https://*.siapnet.com.br/*
// @grant        GM_setValue
// @grant        GM_getValue
// @run-at       document-idle
// ==/UserScript==

(function () {
    'use strict';

    if (window.SiapAutomator && window.SiapAutomator.isInitialized) return;
    window.SiapAutomator = { isInitialized: true };

    const DEFAULT_WEBHOOK = 'http://localhost:5678/webhook/siap-automator';
    const STORAGE_KEY = 'siap_automator_state';

    let stopRequested = false;

    function saveState(data) {
        const existing = GM_getValue(STORAGE_KEY, {});
        const merged = { ...existing, ...data, lastUpdate: Date.now() };
        GM_setValue(STORAGE_KEY, merged);
    }
    function loadState() { return GM_getValue(STORAGE_KEY, {}); }
    function clearState() { GM_setValue(STORAGE_KEY, {}); }

    const logToSidebar = (msg, type = 'info') => {
        const el = document.getElementById('siap-log-output');
        if (!el) return;
        const p = document.createElement('p');
        p.style.margin = '2px 0';
        p.style.color = type === 'error' ? '#ff8888' : type === 'warn' ? '#ffaa00' : type === 'success' ? '#00ff88' : '#00ccff';
        p.textContent = `[${new Date().toLocaleTimeString()}] ${msg}`;
        el.appendChild(p);
        el.scrollTop = el.scrollHeight;
    };

    const wait = (ms) => new Promise(r => setTimeout(r, ms));
    const fireEvent = (target, eventName) => {
        try { target.dispatchEvent(new MouseEvent(eventName, { bubbles: true, cancelable: true })); }
        catch (_) {
            try {
                const evt = document.createEvent('MouseEvents');
                evt.initMouseEvent(eventName, true, true, null, 0, 0, 0, 0, 0, false, false, false, false, 0, null);
                target.dispatchEvent(evt);
            } catch (e2) { logToSidebar(`⚠️ fireEvent falhou: ${e2.message}`, 'warn'); }
        }
    };
    const robustClick = async (el) => {
        if (stopRequested) return;
        el.scrollIntoView({ behavior: 'smooth', block: 'center' });
        await wait(400);
        ['mousedown', 'mouseup', 'click'].forEach(name => fireEvent(el, name));
        const parent = el.closest('a, label, li, td, tr, div[onclick]');
        if (parent && parent !== el) {
            await wait(150);
            ['mousedown', 'mouseup', 'click'].forEach(name => fireEvent(parent, name));
        }
    };
    const typeIntoTextarea = async (el, text) => {
        if (stopRequested) return;
        el.focus();
        el.scrollIntoView({ behavior: 'smooth', block: 'center' });
        await wait(300);
        const nativeSetter = Object.getOwnPropertyDescriptor(window.HTMLTextAreaElement.prototype, 'value')?.set;
        if (nativeSetter) nativeSetter.call(el, text);
        else el.value = text;
        ['input', 'change', 'blur'].forEach(name => el.dispatchEvent(new Event(name, { bubbles: true })));
    };

    // Mapeamento de textareas (original)
    const SECAO_KEYWORDS = {
        metodologias: ['metodologia'],
        avaliacoes: ['avalia'],
    };
    const detectSecao = (text) => {
        const t = text.toLowerCase();
        if (t.includes('metodologia')) return 'metodologias';
        if (t.includes('avalia')) return 'avaliacoes';
        return null;
    };
    const getSectionHeaders = () => {
        const candidates = Array.from(document.querySelectorAll('div, td, th, span, label, b, strong, h2, h3, h4, legend'))
            .filter(el => { const t = el.textContent.trim(); return t.length > 3 && t.length < 80 && detectSecao(t) !== null && el.offsetParent !== null; });
        return candidates.map(el => ({ el, secao: detectSecao(el.textContent.trim()), y: el.getBoundingClientRect().top + window.scrollY })).sort((a,b) => a.y - b.y);
    };
    const buildTextareaMap = () => {
        const headers = getSectionHeaders();
        const textareas = Array.from(document.querySelectorAll('textarea')).filter(ta => ta.offsetParent !== null)
            .map(ta => ({ el: ta, y: ta.getBoundingClientRect().top + window.scrollY })).sort((a,b) => a.y - b.y);
        const map = {};
        for (const ta of textareas) {
            let closest = null;
            for (const h of headers) { if (h.y < ta.y) closest = h; else break; }
            if (closest && !map[closest.secao]) map[closest.secao] = ta.el;
        }
        return map;
    };

    const getItemsFromDoc = (doc) => {
        let items = [];
        const containers = Array.from(doc.querySelectorAll('div[id*="Nodes"]'));
        containers.forEach(container => {
            Array.from(container.querySelectorAll('span, label, a'))
                .map(el => ({ text: el.textContent.trim(), el }))
                .filter(it => it.text.length > 10)
                .forEach(it => items.push(it));
        });
        if (items.length === 0) {
            Array.from(doc.querySelectorAll('span[id*="treeViewt"]'))
                .map(el => ({ text: el.textContent.trim(), el }))
                .filter(it => it.text.length > 15)
                .forEach(it => items.push(it));
        }
        if (items.length === 0) {
            Array.from(doc.querySelectorAll('span, label'))
                .map(el => ({ text: el.textContent.trim(), el }))
                .filter(it => it.text.length > 15 && it.el.offsetParent !== null)
                .forEach(it => items.push(it));
        }
        return items;
    };

    const collectItems = async () => {
        logToSidebar('Buscando itens expandidos...', 'info');
        await wait(1000);
        let all = getItemsFromDoc(document);
        document.querySelectorAll('iframe').forEach(iframe => {
            try { const fd = iframe.contentDocument || iframe.contentWindow.document; all = [...all, ...getItemsFromDoc(fd)]; } catch(_) {}
        });
        const seen = new Set();
        all = all.filter(it => { if(seen.has(it.text)) return false; seen.add(it.text); return true; });
        all.forEach(it => (it.el.style.outline = '2px solid #0056b3'));
        logToSidebar(`${all.length} itens clicáveis encontrados.`, 'success');
        return all;
    };

    const validateAIResponse = (parsed, maxIndex) => {
        if (!parsed || typeof parsed !== 'object') return null;
        let selecionados = parsed.selecionados;
        if (!Array.isArray(selecionados)) selecionados = [];
        if (selecionados.length === 0) return null;
        let validos = selecionados.filter(idx => typeof idx === 'number' && idx >= 0 && idx <= maxIndex);
        if (validos.length === 0) return null;
        let metodologia = (parsed.metodologia || '').trim().substring(0, 1500);
        let avaliacao = (parsed.avaliacao || '').trim().substring(0, 500);
        if (!metodologia) metodologia = 'Explicação na lousa e atividade individual.';
        if (!avaliacao) avaliacao = 'Observação da participação e correção da atividade.';
        return { selecionados: validos, metodologia, avaliacao };
    };

    const clickSalvarProximo = async () => {
        const botoes = document.querySelectorAll('input[type="submit"], button');
        let btn = null;
        for (let b of botoes) {
            if (b.id && b.id.includes('SalvarProximo')) { btn = b; break; }
            if (b.value && (b.value.includes('Salvar') || b.value.includes('Próximo'))) { btn = b; break; }
        }
        if (!btn) btn = document.getElementById('cphFuncionalidade_cphCampos_btnSalvarProximo');
        if (btn) { await robustClick(btn); logToSidebar('✅ Salvar/Próximo clicado.', 'info'); return true; }
        logToSidebar('⚠️ Botão Salvar/Próximo não encontrado.', 'warn');
        return false;
    };

    let isRunning = false;

    const stopAutomation = () => {
        stopRequested = true;
        logToSidebar('🛑 Solicitação de parada recebida. Aguardando fim da etapa atual...', 'warn');
        const btnStop = document.getElementById('siap-stop');
        const btnStart = document.getElementById('siap-start');
        if (btnStop) btnStop.disabled = true;
        if (btnStart) btnStart.disabled = false;
    };

    const runAutomation = async (isAutoStart = false) => {
        if (isRunning) {
            logToSidebar('Automação já em execução.', 'warn');
            return;
        }
        stopRequested = false;
        isRunning = true;

        const btnStart = document.getElementById('siap-start');
        const btnStop = document.getElementById('siap-stop');
        const contextInput = document.getElementById('siap-context').value.trim() || 'Aleatório';
        let qtdTotal = parseInt(document.getElementById('siap-qtd').value) || 1;

        const saved = loadState();
        if (isAutoStart && saved.totalAulas) {
            qtdTotal = saved.totalAulas;
            document.getElementById('siap-qtd').value = qtdTotal;
        }
        if (saved.context && saved.context !== contextInput && !isAutoStart) {
            logToSidebar('⚠️ Contexto mudou. Reiniciando progresso.', 'warn');
            clearState();
        }

        let aulaAtual = saved.currentAula || 1;
        if (aulaAtual > qtdTotal) {
            logToSidebar(`✅ Todas as ${qtdTotal} aulas já foram concluídas.`, 'success');
            isRunning = false;
            btnStart.disabled = false;
            btnStop.disabled = true;
            return;
        }

        if (!isAutoStart) {
            saveState({ currentAula: aulaAtual, context: contextInput, totalAulas: qtdTotal });
        }

        btnStart.disabled = true;
        btnStop.disabled = false;
        logToSidebar(`▶️ Iniciando aula ${aulaAtual} de ${qtdTotal}`, 'info');

        try {
            for (let aula = aulaAtual; aula <= qtdTotal; aula++) {
                if (stopRequested) {
                    logToSidebar('⏸️ Automação interrompida pelo usuário. Progresso salvo.', 'warn');
                    saveState({ currentAula: aula, context: contextInput, totalAulas: qtdTotal });
                    break;
                }
                logToSidebar(`══ Aula ${aula}/${qtdTotal} ══`, 'info');
                await wait(1500);
                if (stopRequested) break;

                const items = await collectItems();
                if (items.length === 0) {
                    logToSidebar('❌ Nenhum item encontrado. Expanda as pastas!', 'error');
                    break;
                }
                const taMap = buildTextareaMap();
                logToSidebar(`Textareas: ${Object.keys(taMap).join(', ')}`, 'info');

                const opcoesNumeradas = items.map((it, idx) => `${idx}: ${it.text}`).join('\n');

                // ─── PROMPT CORRIGIDO: NÃO INSERIR ÍNDICES NOS TEXTOS ─────────
                const prompt = `Você é uma professora brasileira. A aula será aplicada SOMENTE por você, na sala de aula, com recursos simples (lousa, folhas, vídeos, áudios). Nada de convidados ou atividades externas.

Contexto: ${contextInput}

Abaixo estão os itens disponíveis. Os primeiros 3 itens (índices 0, 1, 2) geralmente são HABILIDADES (contêm códigos como (GO-EF05AR07-A)). Os demais são OBJETIVOS DE CONHECIMENTO/CONTEÚDOS.

${opcoesNumeradas}

INSTRUÇÕES:
1. Escolha exatamente 1 item que seja HABILIDADE (índice 0, 1 ou 2).
2. Escolha de 0 a 3 itens que sejam OBJETIVOS (índices a partir de 3).
3. O primeiro índice da lista "selecionados" deve ser a habilidade; os seguintes são os objetivos.
4. "metodologia": Descreva o que você fará em 2-3 linhas, com exemplos concretos. **NÃO mencione números de índice, nem "índice X", nem "item X". Escreva como uma professora real, de forma natural.**
5. "avaliacao": Como avaliará. **Também sem mencionar índices ou posições. Use linguagem natural.**

Responda SOMENTE com JSON:
{
  "selecionados": [indice_da_habilidade, indice_objetivo1, indice_objetivo2, ...],
  "metodologia": "texto da metodologia (sem índices)",
  "avaliacao": "texto da avaliação (sem índices)"
}`;

                const payload = { prompt, context: contextInput, options: items.map(it => it.text), stream: false, format: 'json' };
                logToSidebar(`Enviando ${items.length} itens para a IA...`, 'info');
                const res = await fetch(DEFAULT_WEBHOOK, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(payload) });
                if (!res.ok) { logToSidebar(`❌ HTTP ${res.status}`, 'error'); break; }
                if (stopRequested) break;
                const raw = await res.json();
                let parsed;
                try {
                    let data = raw?.message?.content ?? raw?.response ?? raw;
                    if (typeof data === 'string') { const match = data.match(/\{[\s\S]*\}/); data = JSON.parse(match ? match[0] : data); }
                    parsed = data;
                } catch(e) { logToSidebar(`❌ JSON inválido: ${e.message}`, 'error'); parsed = null; }
                if (stopRequested) break;

                const validation = validateAIResponse(parsed, items.length-1);
                if (!validation) {
                    logToSidebar('⚠️ IA inválida. Usando fallback: apenas primeiro item.', 'warn');
                    const fallbackItem = items[0];
                    if (fallbackItem && !stopRequested) { await robustClick(fallbackItem.el); logToSidebar(`✓ Fallback: ${fallbackItem.text.substring(0,45)}`, 'success'); }
                    const taMet = taMap['metodologias'];
                    if (taMet && !stopRequested) await typeIntoTextarea(taMet, 'Explicação na lousa e atividade individual.');
                    const taAval = taMap['avaliacoes'];
                    if (taAval && !stopRequested) await typeIntoTextarea(taAval, 'Observação da turma.');
                } else {
                    const { selecionados, metodologia, avaliacao } = validation;
                    logToSidebar(`IA selecionou índices: [${selecionados.join(', ')}]`, 'info');
                    for (let idx of selecionados) {
                        if (stopRequested) break;
                        const item = items[idx];
                        if (item) {
                            item.el.style.outline = '3px solid #00ff88';
                            await robustClick(item.el);
                            logToSidebar(`✓ [${idx}] ${item.text.substring(0,45)}`, 'success');
                            await wait(800);
                        } else {
                            logToSidebar(`⚠️ Índice ${idx} inválido`, 'warn');
                        }
                    }
                    const taMet = taMap['metodologias'];
                    if (taMet && metodologia && !stopRequested) { await typeIntoTextarea(taMet, metodologia); logToSidebar('📝 Metodologia preenchida.', 'success'); }
                    const taAval = taMap['avaliacoes'];
                    if (taAval && avaliacao && !stopRequested) { await typeIntoTextarea(taAval, avaliacao); logToSidebar('📝 Avaliação preenchida.', 'success'); }
                }

                if (stopRequested) {
                    logToSidebar('⏸️ Interrompido antes de salvar. Progresso salvo.', 'warn');
                    saveState({ currentAula: aula, context: contextInput, totalAulas: qtdTotal });
                    break;
                }

                logToSidebar('✅ Salvando...', 'warn');
                await wait(3000);
                if (stopRequested) break;
                const savedOk = await clickSalvarProximo();
                if (!savedOk) break;

                if (aula < qtdTotal) {
                    saveState({ currentAula: aula + 1, context: contextInput, totalAulas: qtdTotal });
                    logToSidebar(`Progresso salvo. Próxima aula: ${aula+1}. A página vai recarregar.`, 'info');
                    isRunning = false;
                    btnStart.disabled = false;
                    btnStop.disabled = true;
                    return;
                } else {
                    clearState();
                    logToSidebar('🏁 Todas as aulas concluídas!', 'success');
                }
            }
        } catch(e) {
            logToSidebar(`❌ Erro: ${e.message}`, 'error');
            saveState({ currentAula: aulaAtual, error: e.message });
        } finally {
            isRunning = false;
            btnStart.disabled = false;
            btnStop.disabled = true;
            stopRequested = false;
        }
    };

    const initSidebar = () => {
        if (document.getElementById('siap-sidebar')) return;
        const saved = loadState();
        const resumeHint = saved.currentAula ? ` (retomar aula ${saved.currentAula})` : '';
        document.body.insertAdjacentHTML('beforeend', `
            <div id="siap-sidebar" style="position:fixed; top:0; right:0; width:300px; height:100vh; background:#1a1a2e; color:#eee; border-left:4px solid #0056b3; z-index:100000; display:flex; flex-direction:column; font-family:'Segoe UI',sans-serif; box-shadow:-5px 0 20px rgba(0,0,0,0.5);">
                <div style="background:#0056b3; padding:14px; font-weight:bold;">🤖 SIAP Automator V8.4.6</div>
                <div style="padding:14px; flex:1; display:flex; flex-direction:column; gap:8px;">
                    <label style="font-size:12px; color:#aaa;">📝 Contexto</label>
                    <textarea id="siap-context" rows="3" placeholder="Ex: Aula de Artes Visuais" style="width:100%; padding:8px; background:#0d0d1a; color:#eee; border:1px solid #444; border-radius:5px; resize:none;">${saved.context || ''}</textarea>
                    <label style="font-size:12px; color:#aaa;">🔢 Qtd. Aulas</label>
                    <input type="number" id="siap-qtd" value="${saved.currentAula || 1}" min="1" style="width:100%; padding:8px; background:#0d0d1a; color:#eee; border:1px solid #444; border-radius:5px;">
                    <div style="display:flex; gap:8px;">
                        <button id="siap-start" style="flex:1; background:#27ae60; color:white; padding:13px; border:none; border-radius:6px; cursor:pointer; font-weight:bold;">🚀 Iniciar${resumeHint}</button>
                        <button id="siap-stop" style="flex:1; background:#c0392b; color:white; padding:13px; border:none; border-radius:6px; cursor:pointer; font-weight:bold;" disabled>⏹️ Parar</button>
                    </div>
                    <div style="font-size:10px; color:#666; text-align:center;">✅ 1 Habilidade + até 3 Objetivos<br>📝 Metodologia simples (sala de aula)<br>💾 Automação total + Parar<br>🚫 Sem índices nos textos</div>
                    <div style="font-size:11px; color:#aaa;">📋 Log</div>
                    <div id="siap-log-output" style="background:#0d0d1a; padding:8px; flex:1; overflow-y:auto; font-size:10.5px; border-radius:5px; border:1px solid #333;"></div>
                </div>
            </div>
        `);
        document.getElementById('siap-start').addEventListener('click', () => runAutomation(false));
        document.getElementById('siap-stop').addEventListener('click', stopAutomation);

        if (saved.currentAula && saved.totalAulas && saved.currentAula <= saved.totalAulas) {
            logToSidebar(`Progresso detectado: aula ${saved.currentAula} de ${saved.totalAulas}. Iniciando automaticamente...`, 'info');
            setTimeout(() => runAutomation(true), 2000);
        } else {
            logToSidebar('Pronto. Expanda as pastas e clique em Iniciar.', 'success');
            if (saved.currentAula) logToSidebar(`🔄 Progresso salvo: aula ${saved.currentAula}. Clique em Iniciar para retomar.`, 'warn');
        }
    };
    initSidebar();
})();
